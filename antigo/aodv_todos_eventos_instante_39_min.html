# processamento da tabela de rotamento
start = eventos.iloc[0]['timestamp'] # primeiro evento de alterao da tabela de roteamento
offset = 2.17e6 # offset em milisegundos a partir do instante 'start'
rotas = []
for idx, e in eventos[eventos.timestamp <= start + (offset / 6e4)].iterrows():
    cria = False
    rota = next((r for r in rotas if r['origem'] == e['origem'] and r['destino'] == e['destino']), None)
    if rota == None:
        cria = True
        rota = {'origem': e['origem'], 'destino': e['destino'], 'gateway': e['gateway']}
    if '+' in e['acao'] and cria: # criao de nova rota
        rotas.append(rota)
    elif '*' in e['acao']: # alterao de rota existente
        rota.update({'gateway': e['gateway']})
    else: # removao de rota existente
        rotas.remove(rota)   

# construo do grafo
G = nx.DiGraph()
for h in hosts:
    G.add_node(h)
for r in rotas:
    G.add_edge(r['origem'], r['gateway'])

# clculo das mtricas
delta = start + (offset / 6e4)
minuto = int(delta)
segundo = 60 * (delta - minuto)
dg = []
for n, d in nx.degree(G):
    dg.append(d)
media = mean(dg)
sigma = std(dg)
print('Timestamp: {0:0.0f}:{1:0.0f} (Offset = {2:0.0f} ms)'.format(minuto, segundo, offset))
print('Fracamente conectado: {0}'.format(nx.is_weakly_connected(G)))
print('Fortemente conectado: {0}'.format(nx.is_strongly_connected(G)))
print('Grau mdio: {0}'.format(media))
print('Densidade: {0}'.format(nx.density(G)))
print('Coef. clusterizao mdio: {0}'.format(nx.average_clustering(G)))
try:
    print('Menor caminho mdio: {0}'.format(nx.average_shortest_path_length(G)))
except:
    print('Menor caminho mdio: -')
try:
    print('Dimetro: {0}'.format(nx.diameter(G)))
except:
    print('Dimetro: -')

# processamento da tabela de rotamento com variao no tempo
start = eventos.iloc[0]['timestamp'] # primeiro evento de alterao da tabela de roteamento, em minutos
stop = 60 # condio de parada em minutos
offset = 1000 # offset em milisegundos a partir do instante 'start'

timestamp = []
densidade = []
clustering = []
caminho_medio = []
diametro = []
degree = []
t = start
while t <= stop:
    rotas = []
    for idx, e in eventos[eventos.timestamp <= t].iterrows():
        cria = False
        rota = next((r for r in rotas if r['origem'] == e['origem'] and r['destino'] == e['destino']), None)
        if rota == None:
            cria = True
            rota = {'origem': e['origem'], 'destino': e['destino'], 'gateway': e['gateway']}
        if '+' in e['acao'] and cria: # criao de nova rota
            rotas.append(rota)
        elif '*' in e['acao']: # alterao de rota existente
            rota.update({'gateway': e['gateway']})
        else: # removao de rota existente
            rotas.remove(rota)
    
    # construo do grafo
    G = nx.DiGraph()
    for h in hosts:
        G.add_node(h)
    for r in rotas:
        G.add_edge(r['origem'], r['gateway'])

    # clculo das mtricas
    timestamp.append(t)
    densidade.append(nx.density(G))
    clustering.append(nx.average_clustering(G))
    try:
        caminho_medio.append(nx.average_shortest_path_length(G))
    except:
        caminho_medio.append(None)
    try:
        diametro.append(nx.diameter(G))
    except:
        diametro.append(None)
    dg = []
    for n, d in nx.degree(G):
        dg.append(d)
    degree.append(mean(dg))
    t += offset / 6e4
